A few notable points:

 

         QBW and PBW has multiple application servers.

         These two definitions exist in tnsnames.ora on all the mentioned servers:

BUAWS-KUALI-DB-STAGE.WORLD =

  (DESCRIPTION =

    (ADDRESS_LIST =

      (FAILOVER = ON)

      (LOAD_BALANCE = OFF)

      (ADDRESS = (PROTOCOL = TCP)(HOST = buaws-kuali-db-stage001.bu.edu)(PORT = 1521))

      (ADDRESS = (PROTOCOL = TCP)(HOST = buaws-kuali-db-stage002.bu.edu)(PORT = 1521))

    )

    (CONNECT_DATA =

      (SERVER = DEDICATED)(SERVICE_NAME = kuali_primary)(INSTANCE_ROLE = PRIMARY)

      (FAILOVER_MODE=(TYPE=SELECT)(METHOD=PRECONNECT))

    )

  )
